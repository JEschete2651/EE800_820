\section{Objective}
Assemble the end-to-end inference pipeline: load all three trained classifiers (SF, modulation, emitter \mbox{ID}), stream feature vectors from the \mbox{FPGA} over \UART{} (as established in LH3-F), run all three classifications in parallel, and display live predictions. Verify the total inference latency fits inside the budget established in LH3-D/E. Produce the final model-selection rubric that picks the deployed model per task, consuming every result from LH4-C through LH4-G.

\section{Procedure}

\subsection{Step 1 --- The \texttt{InferenceEngine}}

\begin{lstlisting}[language=Python, caption={\texttt{inference.py} --- loads the chosen model per task and runs all three on each incoming feature vector.}]
import joblib, numpy as np, torch, time
from pathlib import Path

MODEL_CHOICE = {  # Filled in after the rubric in Step 4
    "sf":     "rf",
    "mod":    "mlp",
    "beacon": "cnn",

}

class InferenceEngine:
    def __init__(self, artifacts_dir="artifacts"):
        self.artifacts_dir = Path(artifacts_dir)
        self.models = {}
        self.prep   = {}
        for task, choice in MODEL_CHOICE.items():
            d = self.artifacts_dir / task
            self.prep[task] = joblib.load(d / "preprocessor.pkl")
            self.models[task] = self._load_model(task, choice, d)

    def _load_model(self, task, choice, d):
        if choice in ("rf", "svm"):
            return {"kind": "sk", "model":
                    joblib.load(d / f"{choice}.pkl")["model"]}
        if choice == "mlp":
            from train_mlp import MLP
            ck = torch.load(d / "mlp.pt", map_location="cpu")
            m = MLP(ck["n_features"], ck["n_classes"])
            m.load_state_dict(ck["state_dict"]); m.eval()
            return {"kind": "mlp", "model": m}
        if choice == "cnn":
            from train_cnn import CNN1D
            ck = torch.load(d / "cnn.pt", map_location="cpu")
            m = CNN1D(ck["n_features"], ck["n_classes"])
            m.load_state_dict(ck["state_dict"]); m.eval()
            return {"kind": "cnn", "model": m,
                    "window": ck.get("window", 8)}
        raise ValueError(f"Unknown model choice: {choice}")

    def predict(self, feature_row, sequence_buffer=None):
        """feature_row: 1-D np.array of raw features, same order as preprocessor.
           sequence_buffer: dict task->deque of last N scaled feature vectors."""
        out = {}
        for task, bundle in self.models.items():
            pre = self.prep[task]
            cols = pre["feature_cols"]
            # feature_row must already be in this column order; caller enforces.
            x = feature_row.reshape(1, -1).astype(np.float32)
            x_scaled = pre["scaler"].transform(x)
            if bundle["kind"] == "sk":
                idx = bundle["model"].predict(x_scaled)[0]
                prob = None
                if hasattr(bundle["model"], "predict_proba"):
                    prob = float(bundle["model"].predict_proba(x_scaled).max())
            elif bundle["kind"] == "mlp":
                with torch.no_grad():
                    logits = bundle["model"](torch.from_numpy(x_scaled))
                    probs = torch.softmax(logits, dim=1).numpy()[0]
                    idx  = int(probs.argmax())
                    prob = float(probs.max())
            elif bundle["kind"] == "cnn":
                assert sequence_buffer is not None
                seq = np.stack(list(sequence_buffer[task]))
                seq_t = torch.from_numpy(seq[None, :, :]).float()
                with torch.no_grad():
                    logits = bundle["model"](seq_t)
                    probs = torch.softmax(logits, dim=1).numpy()[0]
                    idx  = int(probs.argmax())
                    prob = float(probs.max())
            label = pre["encoder"].inverse_transform([idx])[0]
            out[task] = {"label": label, "prob": prob}
        return out
\end{lstlisting}

\subsection{Step 2 --- Latency microbenchmark}

\begin{lstlisting}[language=Python, caption={Latency measurement: 1000 warm predictions, report median and 99th percentile.}]
import time, numpy as np
engine = InferenceEngine()
from collections import deque
# Fake a feature row and a filled sequence buffer:
F = len(engine.prep["sf"]["feature_cols"])
seq_buf = {t: deque(np.zeros((8, F), dtype=np.float32), maxlen=8)
           for t in engine.models}
row = np.zeros(F, dtype=np.float32)

# Warm-up
for _ in range(20): engine.predict(row, seq_buf)

latencies = []
for _ in range(1000):
    t0 = time.perf_counter()
    engine.predict(row, seq_buf)
    latencies.append((time.perf_counter() - t0) * 1000)  # ms

latencies = np.asarray(latencies)
print(f"median    = {np.median(latencies):.3f} ms")
print(f"mean      = {latencies.mean():.3f} ms")
print(f"p95       = {np.percentile(latencies, 95):.3f} ms")
print(f"p99       = {np.percentile(latencies, 99):.3f} ms")
print(f"max       = {latencies.max():.3f} ms")
\end{lstlisting}

\begin{tcolorbox}[infobox={Why p99, Not Just Mean?}]
	Garbage collection, Python bytecode JIT warm-up, and occasional OS scheduling hiccups produce a long-tailed latency distribution. The mean can be comfortably under budget while the 99th percentile occasionally breaches it. If p99 $>$ budget, you will see periodic late predictions in the live pipeline --- a problem the mean would have hidden. Report p99 alongside median.
\end{tcolorbox}

\subsection{Step 3 --- Live streaming inference}

\begin{lstlisting}[language=Python, caption={Minimal live loop --- reuses the LH3-F serial reader.}]
from collections import deque
import numpy as np, serial, time
from inference import InferenceEngine

def live_inference(port="COM3", baud=115200):
    engine = InferenceEngine()
    F = len(engine.prep["sf"]["feature_cols"])
    seq_buf = {t: deque(np.zeros((8, F), dtype=np.float32), maxlen=8)
               for t in engine.models}

    with serial.Serial(port, baud, timeout=1.0) as ser:
        while True:
            frame = read_frame(ser)  # from LH3-F collect.py
            if frame is None: continue
            feat_row = frame_to_feature_row(frame, engine.prep["sf"])
            # Update sequence buffer for CNN
            for t in engine.models:
                seq_buf[t].append(engine.prep[t]["scaler"]
                                  .transform(feat_row.reshape(1, -1))[0])
            t0 = time.perf_counter()
            preds = engine.predict(feat_row, seq_buf)
            dt = (time.perf_counter() - t0) * 1000
            print(f"[{time.strftime('%H:%M:%S')}] "
                  f"SF={preds['sf']['label']} "
                  f"(p={preds['sf']['prob']:.2f}) | "
                  f"MOD={preds['mod']['label']} | "
                  f"BEACON={preds['beacon']['label']} | "
                  f"lat={dt:.1f} ms")

if __name__ == "__main__":
    live_inference()
\end{lstlisting}

\subsection{Step 4 --- Final model-selection rubric}
Score each of the four models (RF, SVM, MLP, CNN) per task on a 0--5 scale against each criterion, then weight:

		\textbf{Criterion} & \textbf{Weight} & \textbf{Source of evidence} \\
