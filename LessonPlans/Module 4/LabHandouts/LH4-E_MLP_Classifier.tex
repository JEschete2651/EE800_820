\labheader{4-E}{Multi-Layer Perceptron with Early Stopping and Calibration}

\section{Objective}
Implement, train, and evaluate a small fully-connected neural network ($F\to64\to32\to C$, where $F$ is the per-task feature count and $C$ is the class count) in \texttt{PyTorch} for each of the four classification tasks (SF, modulation, emitter ID, packet type). Use early stopping on the validation loss, apply dropout for regularisation, and compare the \mbox{MLP}'s test accuracy against the random-forest and \mbox{SVM} benchmarks from LH4-C/D. The \mbox{MLP} represents the first ``deep'' model in the project and its job is to demonstrate whether added representational capacity helps on this tabular feature set.

\section{Prerequisites}
\begin{itemize}[leftmargin=2em]
	\item Lab Handout~4-D complete (\texttt{baselines\_manifest.json} and \texttt{svm\_manifest.json} present in \texttt{Code/AIML/}).
	\item \texttt{PyTorch $\geq$ 2.0} installed. \mbox{GPU} is optional --- the model is small enough to train on \mbox{CPU} in seconds per task.
\end{itemize}

\section{Procedure}

\subsection{Step 1 --- Logging, model, and training loop}
\texttt{Code/AIML/mlp\_train.py} follows the same logging pattern as LH4-A through LH4-D. Single-class tasks (\texttt{mod} in the current LoRa-only dataset) are skipped because softmax + cross-entropy reduces to a constant for one class.

class MLP(nn.Module):
    def __init__(self, n_features, n_classes, hidden=(64, 32), p=0.3):
        super().__init__()
        layers = []; prev = n_features
        for h in hidden:
            layers += [nn.Linear(prev, h), nn.ReLU(), nn.Dropout(p)]
            prev = h
        layers.append(nn.Linear(prev, n_classes))
        self.net = nn.Sequential(*layers)

    def forward(self, x):
        return self.net(x)


def load_task(task):
    d = ARTIFACTS_DIR / task
    return (np.load(d/"X_train.npy"), np.load(d/"y_train.npy"),
            np.load(d/"X_val.npy"),   np.load(d/"y_val.npy"),
            np.load(d/"X_test.npy"),  np.load(d/"y_test.npy"),
            joblib.load(d/"preprocessor.pkl"))


def wilson_ci(n_correct, n_total, alpha=0.05):
    if n_total == 0:
        return 0.0, 1.0
    p = n_correct / n_total
    z = norm.ppf(1 - alpha / 2)
    denom = 1 + z * z / n_total
    center = (p + z * z / (2 * n_total)) / denom
    half = z * sqrt(p * (1 - p) / n_total
                    + z * z / (4 * n_total * n_total)) / denom
    return max(0.0, center - half), min(1.0, center + half)


def set_seed(seed):
    torch.manual_seed(seed)
    np.random.seed(seed)
    if torch.cuda.is_available():
        torch.cuda.manual_seed_all(seed)


def train_one(task, seed=42, epochs=200, patience=15, lr=1e-3,
              batch_size=32, verbose=True):
    set_seed(seed)
    X_tr, y_tr, X_va, y_va, X_te, y_te, pre = load_task(task)
    n_classes = len(pre["encoder"].classes_)
    device = "cuda" if torch.cuda.is_available() else "cpu"
    model = MLP(X_tr.shape[1], n_classes).to(device)
    opt   = torch.optim.Adam(model.parameters(), lr=lr)
    loss_fn = nn.CrossEntropyLoss()

    tr_ds = TensorDataset(torch.from_numpy(X_tr).float(),
                          torch.from_numpy(y_tr).long())
    tr_ld = DataLoader(tr_ds, batch_size=batch_size, shuffle=True)

    X_va_t = torch.from_numpy(X_va).float().to(device)
    y_va_t = torch.from_numpy(y_va).long().to(device)

    best_val, best_state, bad = float("inf"), None, 0
    history = {"train_loss": [], "val_loss": [], "val_acc": []}

    for ep in range(epochs):
        model.train()
        total = 0.0
        for xb, yb in tr_ld:
            xb, yb = xb.to(device), yb.to(device)
            opt.zero_grad()
            logits = model(xb)
            loss = loss_fn(logits, yb)
            loss.backward(); opt.step()
            total += loss.item() * len(xb)
        tr_loss = total / len(tr_ds)

        model.eval()
        with torch.no_grad():
            val_logits = model(X_va_t)
            val_loss = loss_fn(val_logits, y_va_t).item()
            val_acc  = (val_logits.argmax(1) == y_va_t).float().mean().item()
        history["train_loss"].append(tr_loss)
        history["val_loss"].append(val_loss)
        history["val_acc"].append(val_acc)

        if val_loss < best_val - 1e-4:
            best_val, bad = val_loss, 0
            best_state = {k: v.clone() for k, v in model.state_dict().items()}
        else:
            bad += 1
            if bad >= patience:
                if verbose:
                    log.info(f"  Early stop at epoch {ep} "
                             f"(best val_loss={best_val:.4f})")
                break

    model.load_state_dict(best_state)
    model.eval()
    with torch.no_grad():
        X_te_t = torch.from_numpy(X_te).float().to(device)
        logits = model(X_te_t)
        y_pred = logits.argmax(1).cpu().numpy()
    n_correct = int((y_pred == y_te).sum())
    acc_te = n_correct / len(y_te) if len(y_te) else 0.0
    return model, history, acc_te, n_correct, len(y_te)
\end{lstlisting}

\begin{lstlisting}[language=Python, caption={Canonical training run (seed 42) per task with single-class skip and Wilson CI.}]
results = {}
for task in ["sf", "mod", "beacon", "pkt"]:
    log.info(f"--- Task: {task} ---")
    X_tr, y_tr, _, _, _, _, pre = load_task(task)
    n_classes = len(pre["encoder"].classes_)
    log.info(f"  Train shape={X_tr.shape}, classes={n_classes}, "
             f"features={pre['feature_cols']}")
    if n_classes < 2:
        log.warning(f"  {task}: only {n_classes} class --- "
                    f"skipping MLP (cross-entropy degenerate for one class)")
        results[task] = {"skipped": True, "reason": "single class in train"}
        continue

    t0 = time.time()
    model, history, acc_te, n_correct, n_total = train_one(task, seed=42)
    elapsed = time.time() - t0
    lo, hi = wilson_ci(n_correct, n_total)
    log.info(f"  MLP test acc = {acc_te:.3f}  [95% CI {lo:.3f}, {hi:.3f}]  "
             f"({n_correct}/{n_total})  trained in {elapsed:.1f}s")

    torch.save({"state_dict": model.state_dict(),
                "n_features": X_tr.shape[1], "n_classes": n_classes,
                "seed": 42, "test_acc": acc_te, "ci": [lo, hi],
                "elapsed_s": elapsed, "history": history},
               ARTIFACTS_DIR / task / "mlp.pt")

    results[task] = {
        "test_acc": acc_te, "ci_lo": lo, "ci_hi": hi,
        "n_correct": n_correct, "n_total": n_total,
        "elapsed_s": elapsed, "n_features": X_tr.shape[1],
        "n_classes": n_classes, "feature_cols": pre["feature_cols"],
    }

\subsection{Step 3 --- Probability calibration}
A classifier that outputs a 90\% probability for a class should be correct 90\% of the time it does so. Plot a reliability diagram (binned empirical frequency vs.\ predicted probability).

\begin{lstlisting}[language=Python]
def calibration_plot(model, X_te, y_te, task):
    model.eval()
    device = next(model.parameters()).device
    X_t = torch.from_numpy(X_te).float().to(device)
    with torch.no_grad():
        probs = torch.softmax(model(X_t), dim=1).cpu().numpy()
    pred     = probs.argmax(axis=1)
    max_prob = probs.max(axis=1)
    correct  = (pred == y_te).astype(int)
    # Uniform bins over [0, 1]; quantile bins waste resolution at the
    # high-confidence end where most predictions live.
    n_bins = 8
    frac_pos, mean_pred = calibration_curve(
        correct, max_prob, n_bins=n_bins, strategy="uniform")
    fig, ax = plt.subplots(figsize=(5, 4.5))
    ax.plot([0, 1], [0, 1], "--", color="gray", label="perfect")
    ax.plot(mean_pred, frac_pos, "o-", color="steelblue", label="MLP")
    ax.set_xlabel("Predicted confidence (top-1 prob)")
    ax.set_ylabel("Empirical accuracy")
    ax.set_title(f"Reliability diagram --- {task}")
    ax.set_xlim(0, 1); ax.set_ylim(0, 1); ax.legend(loc="lower right")
    fig.tight_layout()
    out = FIGS_DIR / f"mlp_reliability_{task}.png"
    fig.savefig(out, dpi=150)
    plt.close(fig)
    log.info(f"  Saved {out}")
    return list(zip(mean_pred.tolist(), frac_pos.tolist()))


for task, res in results.items():
    if res.get("skipped"):
        continue
    pkl = torch.load(ARTIFACTS_DIR / task / "mlp.pt", weights_only=False)
    _, _, _, _, X_te, y_te, _ = load_task(task)
    model = MLP(pkl["n_features"], pkl["n_classes"])
    model.load_state_dict(pkl["state_dict"])
    res["calibration"] = calibration_plot(model, X_te, y_te, task)
\end{lstlisting}

If the curve sits above the diagonal, the model is \emph{under}-confident (real accuracy is higher than reported probability); if below, it is \emph{over}-confident. For deployment, over-confidence is the dangerous case.

\begin{tcolorbox}[infobox={Why Calibration Matters for Spectrum Monitoring}]
	The downstream consumer of this classifier's output is a decision process that may weight SF-probability estimates when doing multi-beacon tracking. An over-confident classifier will repeatedly assert wrong labels with high probability, pushing the tracking system toward false associations. A well-calibrated classifier --- even one with slightly lower raw accuracy --- produces more useful probability estimates for downstream fusion.
\end{tcolorbox}

\subsection{Step 4 --- Random-seed robustness}
Re-train the \mbox{MLP} for each non-skipped task with seeds 0, 1, 2, 3 (in addition to the canonical seed-42 run from Step 1). Report mean and standard deviation of the test accuracy per task. If the std exceeds $\sim$0.02 on any task, that model is unstable; consider more regularisation or additional data augmentation before promoting it. The Step 1 \texttt{mlp.pt} (seed 42) is left untouched --- the seed sweep is for measurement, not replacement.

\begin{lstlisting}[language=Python]
log.info("=== Seed robustness ===")
for task, res in results.items():
    if res.get("skipped"):
        continue
    accs = []
    for seed in range(4):
        _, _, acc, _, _ = train_one(task, seed=seed, verbose=False)
        accs.append(acc)
    mean = float(np.mean(accs))
    std  = float(np.std(accs))
    log.info(f"  {task:<8} seeds 0-3: accs={[round(a, 3) for a in accs]}  "
             f"mean={mean:.3f} std={std:.3f}")
    if std > 0.02:
        log.warning(f"  {task}: seed std {std:.3f} > 0.02 --- "
                    f"model is unstable; consider more regularisation")
    res["seed_sweep"] = {"seeds": list(range(4)), "accs": accs,
                         "mean": mean, "std": std}
\end{lstlisting}
