\labheader{4-B}{Preprocessing Pipeline, Feature Selection, and Label-Leakage Prevention}

\section{Objective}
Convert the validated dataset from LH4-A into model-ready matrices: drop sentinel rows, encode labels, select task-specific feature sets that avoid label leakage, and fit a \texttt{StandardScaler} that downstream classifiers will reuse. The deliverable is a reusable \texttt{Preprocessor} class and three saved artifacts (\texttt{X\_train.npy}, \texttt{y\_train.npy}, \texttt{scaler.pkl}) per classification task.

\section{Prerequisites}
\begin{itemize}[leftmargin=2em]
	\item Lab Handout~4-A complete (validated dataset, \texttt{validation\_manifest.json}).
	\item Python with \texttt{scikit-learn} and \texttt{joblib} installed.
\end{itemize}

\section{Background: Why Three Feature Sets, Not One?}
Module~4 trains three classifiers: spreading-factor detection, modulation classification, and emitter identification. Each has a different list of columns that are \emph{forbidden} because they would trivially reveal the label:

\begin{table}[h!]
	\centering
	\renewcommand{\arraystretch}{1.3}
	\begin{tabularx}{\textwidth}{l l X}
		\toprule
		\textbf{Task} & \textbf{Label} & \textbf{Columns Forbidden (Would Leak Label)} \\
		\midrule
		SF Detection & \texttt{label\_sf} & \texttt{spread\_factor} (self-reported copy of the label) \\
		Modulation   & \texttt{label\_mod} & \texttt{mod\_code}, \texttt{spread\_factor} (SF is 0 for \mbox{FSK}) \\
		Emitter ID   & \texttt{label\_beacon} & \texttt{beacon\_id} (self-reported copy of the label) \\
		\bottomrule
	\end{tabularx}
	\caption{Label-leakage table. Any of these forbidden columns left in the feature matrix will push test accuracy to $\sim$100\% --- not because the classifier works, but because it is reading the answer.}
\end{table}

\begin{tcolorbox}[warning={The Silent Leak}]
	Label leakage rarely produces an error message. It produces an impressive accuracy number. The only way to catch it is to enumerate your forbidden columns \emph{before} training, and the only way to keep that list current is to derive it mechanically from the data dictionary. If you add a new feature to the \mbox{CSV} later, revisit this table.
\end{tcolorbox}

\section{Procedure}

\subsection{Step 1 --- Drop sentinel rows}
Rows with \texttt{obs\_count == 1} have undefined \texttt{inter\_arrival\_ms}, \texttt{rssi\_delta}, and \texttt{priority\_score} (nothing to compare the first observation against). Drop them globally before any feature selection.

\begin{lstlisting}[language=Python]
import pandas as pd
df = pd.read_csv("merged_dataset.csv")
n0 = len(df)
df = df[df["obs_count"] > 1].reset_index(drop=True)
print(f"Dropped {n0 - len(df)} sentinel rows ({(n0 - len(df))/n0:.1%})")
\end{lstlisting}

\subsection{Step 2 --- Build the \texttt{Preprocessor} class}

\begin{lstlisting}[language=Python, caption={\texttt{preprocess.py} --- the reusable preprocessing class used by every Module~4 classifier.}]
import joblib
import numpy as np
import pandas as pd
from sklearn.preprocessing import StandardScaler, LabelEncoder
from pathlib import Path

FEATURE_COLS_BASE = [
    "rssi", "snr", "inter_arrival_ms", "rssi_delta",
    "obs_count", "priority_score", "distance_m",
]

FORBIDDEN = {
    "sf":     ["spread_factor"],
    "mod":    ["mod_code", "spread_factor"],
    "beacon": ["beacon_id"],
}

LABEL_COL = {
    "sf":     "label_sf",
    "mod":    "label_mod",
    "beacon": "label_beacon",
}

class Preprocessor:
    def __init__(self, task):
        assert task in FORBIDDEN, f"Unknown task: {task}"
        self.task = task
        self.feature_cols = [c for c in FEATURE_COLS_BASE
                             if c not in FORBIDDEN[task]]
        self.scaler = StandardScaler()
        self.encoder = LabelEncoder()

    def fit_transform(self, df):
        tr = df[df["split"] == "train"]
        X_tr = tr[self.feature_cols].to_numpy(dtype=np.float32)
        y_tr = self.encoder.fit_transform(tr[LABEL_COL[self.task]])
        X_tr = self.scaler.fit_transform(X_tr)
        return X_tr, y_tr

    def transform(self, df, split):
        sp = df[df["split"] == split]
        X = sp[self.feature_cols].to_numpy(dtype=np.float32)
        y = self.encoder.transform(sp[LABEL_COL[self.task]])
        X = self.scaler.transform(X)
        return X, y

    def save(self, out_dir):
        out = Path(out_dir)
        out.mkdir(parents=True, exist_ok=True)
        joblib.dump({"scaler": self.scaler,
                     "encoder": self.encoder,
                     "feature_cols": self.feature_cols,
                     "task": self.task}, out / "preprocessor.pkl")
\end{lstlisting}

\subsection{Step 3 --- Generate task-specific matrices}

\begin{lstlisting}[language=Python, caption={Run preprocessing for all three tasks.}]
from preprocess import Preprocessor
import numpy as np

for task in ["sf", "mod", "beacon"]:
    pre = Preprocessor(task)
    X_tr, y_tr = pre.fit_transform(df)
    X_va, y_va = pre.transform(df, "val")
    X_te, y_te = pre.transform(df, "test")

    print(f"\n=== Task: {task} ===")
    print(f"Features: {pre.feature_cols}")
    print(f"Classes:  {list(pre.encoder.classes_)}")
    print(f"Shapes:   X_tr={X_tr.shape}, X_va={X_va.shape}, X_te={X_te.shape}")

    out = f"artifacts/{task}"
    pre.save(out)
    np.save(f"{out}/X_train.npy", X_tr); np.save(f"{out}/y_train.npy", y_tr)
    np.save(f"{out}/X_val.npy",   X_va); np.save(f"{out}/y_val.npy",   y_va)
    np.save(f"{out}/X_test.npy",  X_te); np.save(f"{out}/y_test.npy",  y_te)
\end{lstlisting}

\subsection{Step 4 --- Verify the scaler}
A fitted \texttt{StandardScaler} on the training set produces zero mean and unit variance per feature:

\begin{lstlisting}[language=Python]
for task in ["sf", "mod", "beacon"]:
    X_tr = np.load(f"artifacts/{task}/X_train.npy")
    mu, sd = X_tr.mean(axis=0), X_tr.std(axis=0)
    print(f"{task}: mean max|mu|={np.abs(mu).max():.3e}, "
          f"std range=[{sd.min():.3f}, {sd.max():.3f}]")
    assert np.abs(mu).max() < 1e-5
    assert np.all(np.abs(sd - 1) < 1e-5)
\end{lstlisting}

Validation and test matrices will \emph{not} have exactly zero-mean/unit-variance because the scaler was fit on training data only --- that is correct behaviour and the whole reason \texttt{fit\_transform} is called on train while plain \texttt{transform} is called on val and test.

\begin{tcolorbox}[infobox={When Not to Scale}]
	Tree-based classifiers (decision tree, random forest, extra-trees) are invariant to monotonic transformations of single features. Scaling them does not hurt but adds needless overhead. LH4-C and LH4-D still load the scaled matrices because using a single set of artifacts keeps the experiment repeatable across every model, but do \emph{not} quote a decision-tree result as ``better with scaling'' --- it is identical up to floating-point noise.
\end{tcolorbox}

\subsection{Step 5 --- Leakage sanity check}
Train a quick logistic regression on each task using \emph{only} the forbidden columns, and confirm accuracy is near 100\%. This demonstrates that the forbidden list is correct: leaving those columns in would trivialise the problem.

\begin{lstlisting}[language=Python]
from sklearn.linear_model import LogisticRegression
from sklearn.preprocessing import StandardScaler

for task, bad in FORBIDDEN.items():
    tr = df[df["split"] == "train"]
    te = df[df["split"] == "test"]
    # label_mod is a string; encode it the same way the Preprocessor did
    from sklearn.preprocessing import LabelEncoder
    enc = LabelEncoder()
    y_tr = enc.fit_transform(tr[LABEL_COL[task]])
    y_te = enc.transform(te[LABEL_COL[task]])
    X_tr = tr[bad].to_numpy(dtype=float)
    X_te = te[bad].to_numpy(dtype=float)
    sc = StandardScaler().fit(X_tr)
    clf = LogisticRegression(max_iter=1000).fit(sc.transform(X_tr), y_tr)
    acc = clf.score(sc.transform(X_te), y_te)
    print(f"{task} (forbidden-only features): test acc = {acc:.3f}")
\end{lstlisting}

A result below $\sim$0.95 means the forbidden list is incomplete for that task; revisit and add the missing column.

\section{Verification Checklist}
\begin{itemize}[leftmargin=2em]
	\item[$\square$] Sentinel rows dropped; count matches LH4-A audit summary.
	\item[$\square$] \texttt{Preprocessor} class saves and reloads without error.
	\item[$\square$] All three tasks have six saved arrays (\texttt{X\_train}, \texttt{X\_val}, \texttt{X\_test} and three matching \texttt{y\_}).
	\item[$\square$] Training-set scaler produces zero mean / unit variance within $10^{-5}$.
	\item[$\square$] Leakage sanity check: each task's forbidden-only classifier scores $\geq$0.95.
\end{itemize}
