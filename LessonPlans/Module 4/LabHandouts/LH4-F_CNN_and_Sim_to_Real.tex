\labheader{4-F}{1-D CNN over Feature Sequences and the Sim-to-Real Gap Experiment}

\section{Objective}
Build and train a 1-D convolutional network that consumes short \emph{sequences} of feature vectors (rather than single vectors like the \mbox{MLP} in LH4-E), then run the defining sim-to-real transfer experiment: train on simulator data, evaluate on hardware data, and quantify the accuracy gap. This is the capstone machine-learning result that closes the loop on the distribution-shift analysis delivered in LH3-H.

\section{Prerequisites}
\begin{itemize}[leftmargin=2em]
	\item Lab Handout~4-E complete (\mbox{MLP} numbers available for comparison).
	\item Simulator-produced feature \mbox{CSV}s from the earlier portion of the course (these are the ones used in the LH3-H distribution-shift analysis). If only one simulator dataset exists, that is sufficient --- this handout needs one simulator set and one hardware set.
\end{itemize}

\section{Background: Why a 1-D CNN on Tabular Features?}
A single feature vector captures one instant. A short window of consecutive vectors --- say the last eight packets from the same beacon --- captures the \emph{temporal texture} of the transmission: burstiness, inter-arrival regularity, \mbox{RSSI} drift, \mbox{SNR} stability. A 1-D \mbox{CNN} over that sequence axis can learn filters that respond to this temporal structure. The \mbox{MLP} cannot, because it sees one vector at a time.

\begin{tcolorbox}[infobox={Sequence Construction}]
	For each row $r$ in the dataset, build an $8 \times F$ input by taking the eight most recent packets with the same \texttt{beacon\_id} (or fewer if eight are not yet available; left-pad with zeros). This produces one sequence per row while preserving the label on $r$. The sequences are \emph{within-campaign} only --- do not straddle campaign boundaries, because that would force the model to learn transitions between very different channel conditions.
\end{tcolorbox}

\section{Procedure}

\subsection{Step 1 --- Build sliding-window sequences}

\begin{lstlisting}[language=Python, caption={\texttt{build\_sequences.py} --- produce the $8 \times F$ tensors from the preprocessed artifacts.}]
import numpy as np, pandas as pd, joblib
from pathlib import Path

WINDOW = 8

def build_sequences(task):
    df = pd.read_csv("merged_dataset.csv")
    df = df[df["obs_count"] > 1].reset_index(drop=True)
    pre = joblib.load(f"artifacts/{task}/preprocessor.pkl")
    feats = pre["feature_cols"]
    # Scale all rows with the trained scaler:
    X_all = pre["scaler"].transform(df[feats].to_numpy(dtype=np.float32))
    y_all = pre["encoder"].transform(df[{
        "sf": "label_sf", "mod": "label_mod", "beacon": "label_beacon"
    }[task]])

    seqs, labels, splits = [], [], []
    for (campaign, beacon), grp in df.groupby(["campaign_id", "beacon_id"]):
        idxs = grp.index.to_numpy()
        for i, r in enumerate(idxs):
            lo = max(0, i - WINDOW + 1)
            window_idxs = idxs[lo:i + 1]
            window = X_all[window_idxs]
            # Left-pad with zeros if short
            if len(window) < WINDOW:
                pad = np.zeros((WINDOW - len(window), window.shape[1]),
                               dtype=np.float32)
                window = np.concatenate([pad, window], axis=0)
            seqs.append(window)
            labels.append(y_all[r])
            splits.append(df.at[r, "split"])

    seqs   = np.stack(seqs)
    labels = np.asarray(labels, dtype=np.int64)
    splits = np.asarray(splits)
    for name in ["train", "val", "test"]:
        mask = splits == name
        np.save(f"artifacts/{task}/Xs_{name}.npy", seqs[mask])
        np.save(f"artifacts/{task}/ys_{name}.npy", labels[mask])
        print(f"[{task}] {name}: {mask.sum()} sequences of "
              f"shape {seqs.shape[1:]}")

for t in ["sf", "mod", "beacon"]:
    build_sequences(t)
\end{lstlisting}

\subsection{Step 2 --- CNN-1D model}

\begin{lstlisting}[language=Python, caption={\texttt{train\_cnn.py} --- small 1-D CNN with global average pooling.}]
import torch, torch.nn as nn, numpy as np
from torch.utils.data import TensorDataset, DataLoader

class CNN1D(nn.Module):
    def __init__(self, n_features, n_classes):
        super().__init__()
        self.net = nn.Sequential(
            # Input is (B, F, T); Conv1d convolves along T
            nn.Conv1d(n_features, 16, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.Conv1d(16, 32, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.AdaptiveAvgPool1d(1),   # global average pool over T
            nn.Flatten(),
            nn.Linear(32, 32), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(32, n_classes),
        )

    def forward(self, x):
        # x comes in as (B, T, F); swap to (B, F, T)
        x = x.transpose(1, 2)
        return self.net(x)

def train_cnn(task, epochs=200, patience=15, lr=1e-3, batch_size=32):
    Xs_tr = np.load(f"artifacts/{task}/Xs_train.npy")
    ys_tr = np.load(f"artifacts/{task}/ys_train.npy")
    Xs_va = np.load(f"artifacts/{task}/Xs_val.npy")
    ys_va = np.load(f"artifacts/{task}/ys_val.npy")
    Xs_te = np.load(f"artifacts/{task}/Xs_test.npy")
    ys_te = np.load(f"artifacts/{task}/ys_test.npy")

    n_classes = int(ys_tr.max()) + 1
    device = "cuda" if torch.cuda.is_available() else "cpu"
    model = CNN1D(Xs_tr.shape[2], n_classes).to(device)
    opt = torch.optim.Adam(model.parameters(), lr=lr)
    loss_fn = nn.CrossEntropyLoss()
    ds = TensorDataset(torch.from_numpy(Xs_tr), torch.from_numpy(ys_tr))
    ld = DataLoader(ds, batch_size=batch_size, shuffle=True)
    X_va_t = torch.from_numpy(Xs_va).to(device)
    y_va_t = torch.from_numpy(ys_va).to(device)

    best_val, best_state, bad = float("inf"), None, 0
    for ep in range(epochs):
        model.train()
        for xb, yb in ld:
            xb, yb = xb.to(device), yb.to(device)
            opt.zero_grad()
            loss = loss_fn(model(xb), yb)
            loss.backward(); opt.step()
        model.eval()
        with torch.no_grad():
            v_logits = model(X_va_t)
            v_loss = loss_fn(v_logits, y_va_t).item()
        if v_loss < best_val - 1e-4:
            best_val = v_loss; best_state = {k: v.clone()
                for k, v in model.state_dict().items()}; bad = 0
        else:
            bad += 1
            if bad >= patience: break

    model.load_state_dict(best_state)
    model.eval()
    with torch.no_grad():
        X_te_t = torch.from_numpy(Xs_te).to(device)
        y_te_t = torch.from_numpy(ys_te).to(device)
        acc = (model(X_te_t).argmax(1) == y_te_t).float().mean().item()
    torch.save({"state_dict": model.state_dict(),
                "n_features": Xs_tr.shape[2], "n_classes": n_classes,
                "test_acc": acc}, f"artifacts/{task}/cnn.pt")
    print(f"[{task}] CNN test acc = {acc:.3f}")
    return model, acc

for t in ["sf", "mod", "beacon"]:
    train_cnn(t)
\end{lstlisting}

\subsection{Step 3 --- The sim-to-real transfer experiment}
Define:
\begin{itemize}[leftmargin=2em]
	\item $\text{Acc}(\text{hw}\to\text{hw})$: model trained on hardware data, evaluated on held-out hardware test set.
	\item $\text{Acc}(\text{sim}\to\text{hw})$: model trained on simulator data only, evaluated on the same hardware test set.
	\item The sim-to-real gap: $\Delta = \text{Acc}(\text{hw}\to\text{hw}) - \text{Acc}(\text{sim}\to\text{hw})$.
\end{itemize}

A larger $\Delta$ means the simulator's distribution is further from reality. Report $\Delta$ for each task.

\begin{lstlisting}[language=Python, caption={Sim-to-real evaluation --- full reference implementation.}]
# Requires: numpy, pandas, torch, scikit-learn. Assumes the Preprocessor
# from LH4-B is importable as `build_preprocessor`, and CNN1D is defined
# earlier in this handout. simulator_dataset.csv has the same schema as
# merged_dataset.csv (LH3-H contract).

import numpy as np
import pandas as pd
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, TensorDataset

DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")
LABEL_COL = {"sf": "label_sf", "mod": "mod_code_label", "beacon": "label_beacon"}
WIN = 8  # sliding window length, matches Step 1

def _load_windows(csv_path, task, scaler=None, fit_scaler=False):
    df = pd.read_csv(csv_path)
    # Filter to rows the preprocessor deems usable (drop sentinels etc.)
    df = df[df["obs_count"] > 1].reset_index(drop=True)
    feat_cols = ["rssi", "snr", "inter_arrival_ms", "rssi_delta",
                 "obs_count", "priority_score", "spread_factor"]
    X = df[feat_cols].to_numpy(dtype=np.float32)
    y = df[LABEL_COL[task]].to_numpy()
    if fit_scaler:
        scaler = build_preprocessor()           # StandardScaler from LH4-B
        X = scaler.fit_transform(X)
    else:
        X = scaler.transform(X)
    # Build sliding windows WITHIN run_id only (no cross-run contamination)
    Xs, ys = [], []
    for rid, grp in df.groupby("run_id"):
        idx = grp.index.to_numpy()
        if len(idx) < WIN: continue
        for i in range(len(idx) - WIN + 1):
            sl = idx[i:i + WIN]
            Xs.append(X[sl]); ys.append(y[sl[-1]])
    return np.stack(Xs), np.asarray(ys), scaler

def _train(model, X, y, epochs=20, batch=64, lr=1e-3):
    model = model.to(DEVICE)
    ds = TensorDataset(torch.from_numpy(X).float(),
                       torch.from_numpy(y).long())
    dl = DataLoader(ds, batch_size=batch, shuffle=True)
    opt = torch.optim.Adam(model.parameters(), lr=lr)
    loss_fn = nn.CrossEntropyLoss()
    model.train()
    for _ in range(epochs):
        for xb, yb in dl:
            xb, yb = xb.to(DEVICE), yb.to(DEVICE)
            opt.zero_grad()
            loss_fn(model(xb), yb).backward()
            opt.step()
    return model

@torch.no_grad()
def _accuracy(model, X, y):
    model.eval()
    xb = torch.from_numpy(X).float().to(DEVICE)
    yb = torch.from_numpy(y).long().to(DEVICE)
    pred = model(xb).argmax(dim=1)
    return (pred == yb).float().mean().item()

def sim2real(task, model_fn, train_csv, eval_csv):
    """Train a fresh model on train_csv, evaluate on eval_csv's test split.

    Preprocessing policy (fair-comparison rule): fit the scaler on the
    training CSV, then apply it to the evaluation CSV. This keeps train-time
    statistics out of the eval path for sim->hw runs.
    """
    # Fit scaler on training data; transform eval data with it.
    X_tr, y_tr, scaler = _load_windows(train_csv, task, fit_scaler=True)
    X_ev, y_ev, _      = _load_windows(eval_csv,  task, scaler=scaler)
    # When eval_csv == train_csv, subset to the held-out split to avoid
    # evaluating on training rows.
    if train_csv == eval_csv:
        df = pd.read_csv(eval_csv)
        df = df[df["obs_count"] > 1].reset_index(drop=True)
        test_mask = (df["split"] == "test").to_numpy()
        # Rebuild windows using only test rows (simple approximation):
        # drop windows whose final row is not in test.
        final_is_test = np.array([test_mask[i + WIN - 1]
                                  for i in range(len(X_ev))])
        X_ev, y_ev = X_ev[final_is_test], y_ev[final_is_test]
    n_classes = int(max(y_tr.max(), y_ev.max())) + 1
    model = model_fn(in_ch=X_tr.shape[2], n_classes=n_classes, win=WIN)
    model = _train(model, X_tr, y_tr)
    return _accuracy(model, X_ev, y_ev)

results = []
for task in ["sf", "mod", "beacon"]:
    acc_hw  = sim2real(task, CNN1D, "merged_dataset.csv",
                                     "merged_dataset.csv")
    acc_sim = sim2real(task, CNN1D, "simulator_dataset.csv",
                                     "merged_dataset.csv")
    delta = acc_hw - acc_sim
    results.append((task, acc_hw, acc_sim, delta))
    print(f"[{task}] hw->hw = {acc_hw:.3f}, sim->hw = {acc_sim:.3f}, "
          f"Delta = {delta:.3f}")
\end{lstlisting}

\begin{tcolorbox}[warning={Fair-Comparison Rule}]
	The simulator and hardware datasets must be processed with the \emph{same} preprocessing pipeline. Fit the \texttt{StandardScaler} on the simulator training set for the sim$\to$hw experiment, then apply that same scaler to the hardware test set --- the scaler is part of the model. If you fit the scaler on hardware training data for both experiments, you are ``leaking'' hardware statistics into the sim$\to$hw model, which understates the gap.
\end{tcolorbox}

\subsection{Step 5 --- Mitigation: fine-tuning}
Take the model trained on simulator data, freeze all layers except the final \texttt{Linear}, and fine-tune on a small fraction (10--20\%) of the hardware training set. Measure the new accuracy on the hardware test set. Expected outcome: most of the sim-to-real gap closes, demonstrating that the backbone learned useful representations but the classifier head was mis-calibrated to the hardware distribution.

\begin{lstlisting}[language=Python]
def finetune_last_layer(task, sim_model, hw_train_csv, hw_eval_csv,
                        fraction=0.10, epochs=5, batch=32, lr=1e-4):
    # Freeze backbone, train only the final Linear layer
    for p in sim_model.parameters():       p.requires_grad = False
    for p in sim_model.net[-1].parameters(): p.requires_grad = True

    X_ft, y_ft, scaler = _load_windows(hw_train_csv, task, fit_scaler=True)
    # Take a random fraction of the training windows (reproducible seed)
    rng = np.random.default_rng(seed=42)
    keep = rng.choice(len(X_ft), size=int(fraction * len(X_ft)),
                      replace=False)
    X_ft, y_ft = X_ft[keep], y_ft[keep]

    sim_model = sim_model.to(DEVICE)
    ds = TensorDataset(torch.from_numpy(X_ft).float(),
                       torch.from_numpy(y_ft).long())
    dl = DataLoader(ds, batch_size=batch, shuffle=True)
    opt = torch.optim.Adam([p for p in sim_model.parameters()
                            if p.requires_grad], lr=lr)
    loss_fn = nn.CrossEntropyLoss()
    sim_model.train()
    for _ in range(epochs):
        for xb, yb in dl:
            xb, yb = xb.to(DEVICE), yb.to(DEVICE)
            opt.zero_grad()
            loss_fn(sim_model(xb), yb).backward()
            opt.step()

    # Evaluate on the hardware held-out test split
    X_ev, y_ev, _ = _load_windows(hw_eval_csv, task, scaler=scaler)
    return _accuracy(sim_model, X_ev, y_ev)

# Run for each task, reusing the sim-trained model from the sim2real call
for task in ["sf", "mod", "beacon"]:
    # Rebuild the sim-trained model exactly as in sim2real above
    X_tr, y_tr, scaler = _load_windows("simulator_dataset.csv", task,
                                        fit_scaler=True)
    n_cls = int(y_tr.max()) + 1
    sim_model = _train(CNN1D(in_ch=X_tr.shape[2], n_classes=n_cls, win=WIN),
                       X_tr, y_tr)
    acc_finetune = finetune_last_layer(task, sim_model,
                                       "merged_dataset.csv",
                                       "merged_dataset.csv")
    print(f"[{task}] sim + HW-finetune test acc = {acc_finetune:.3f}")
\end{lstlisting}
