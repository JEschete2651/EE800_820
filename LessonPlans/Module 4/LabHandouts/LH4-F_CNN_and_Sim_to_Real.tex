\labheader{4-F}{1-D CNN over Feature Sequences and the Clean$\to$Contended Operating-Condition Shift Experiment}

\section{Objective}
Build and train a 1-D convolutional network that consumes short \emph{sequences} of feature vectors (rather than single vectors like the \mbox{MLP} in LH4-E), then run the defining operating-condition shift experiment: train on the clean regime (Module~3 campaigns C1--C4: single-station Target~A only, DATA-only) and evaluate on the contended regime (C5: cooperative-target ping-pong with collisions and interleaved Threat PAUSE bursts). Both regimes come from the same hardware --- this is a within-hardware distribution-shift study, the most honest test of generalisation the framework can produce. The accuracy gap is the framework's capstone machine-learning result.

\section{Prerequisites}
\begin{itemize}[leftmargin=2em]
	\item Lab Handout~4-E complete (\mbox{MLP} numbers available for comparison on all four tasks).
	\item Module~3 dataset (\texttt{merged\_dataset.csv}) contains the \texttt{regime} column tagging each row as \texttt{clean} (C1--C4) or \texttt{contended} (C5).
\end{itemize}

\section{Background: Why a 1-D CNN on Tabular Features?}
A single feature vector captures one instant. A short window of consecutive vectors --- say the last eight packets from the same station --- captures the \emph{temporal texture} of the transmission: burstiness, inter-arrival regularity, \mbox{RSSI} drift, \mbox{SNR} stability, the cadence difference between DATA / ACK / PAUSE rows. A 1-D \mbox{CNN} over that sequence axis can learn filters that respond to this temporal structure. The \mbox{MLP} cannot, because it sees one vector at a time. The packet-type task is expected to benefit most from the windowed view, because DATA / ACK / PAUSE have very different inter-arrival signatures.

\begin{tcolorbox}[infobox={Sequence Construction}]
	For each row $r$ in the dataset, build an $8 \times F$ input by taking the eight most recent packets with the same \texttt{beacon\_id} (or fewer if eight are not yet available; left-pad with zeros). This produces one sequence per row while preserving the label on $r$. The sequences are \emph{within-run} only --- do not straddle \texttt{run\_id} boundaries, because that would force the model to learn transitions between very different channel conditions.
\end{tcolorbox}

\section{Procedure}

\subsection{Step 1 --- Build sliding-window sequences}

\begin{lstlisting}[language=Python, caption={\texttt{build\_sequences.py} --- produce the $8 \times F$ tensors from the preprocessed artifacts.}]
import numpy as np, pandas as pd, joblib
from pathlib import Path

WINDOW = 8

def build_sequences(task):
    df = pd.read_csv("merged_dataset.csv")
    df = df[df["obs_count"] > 1].reset_index(drop=True)
    pre = joblib.load(f"artifacts/{task}/preprocessor.pkl")
    feats = pre["feature_cols"]
    # Scale all rows with the trained scaler:
    X_all = pre["scaler"].transform(df[feats].to_numpy(dtype=np.float32))
    label_col = {
        "sf": "label_sf", "mod": "label_mod",
        "beacon": "label_beacon", "pkt": "label_pkt",
    }[task]
    y_all = pre["encoder"].transform(df[label_col])

    seqs, labels, splits, regimes = [], [], [], []
    # Group by run_id (not campaign_id) to avoid cross-run contamination:
    # each run_id is one collect.py invocation under stable channel state.
    for rid, grp in df.groupby("run_id"):
        idxs = grp.index.to_numpy()
        for i, r in enumerate(idxs):
            lo = max(0, i - WINDOW + 1)
            window_idxs = idxs[lo:i + 1]
            window = X_all[window_idxs]
            # Left-pad with zeros if short
            if len(window) < WINDOW:
                pad = np.zeros((WINDOW - len(window), window.shape[1]),
                               dtype=np.float32)
                window = np.concatenate([pad, window], axis=0)
            seqs.append(window)
            labels.append(y_all[r])
            splits.append(df.at[r, "split"])
            regimes.append(df.at[r, "regime"])

    seqs    = np.stack(seqs)
    labels  = np.asarray(labels, dtype=np.int64)
    splits  = np.asarray(splits)
    regimes = np.asarray(regimes)
    for name in ["train", "val", "test"]:
        mask = splits == name
        np.save(f"artifacts/{task}/Xs_{name}.npy",      seqs[mask])
        np.save(f"artifacts/{task}/ys_{name}.npy",      labels[mask])
        np.save(f"artifacts/{task}/regs_{name}.npy",    regimes[mask])
        print(f"[{task}] {name}: {mask.sum()} sequences of "
              f"shape {seqs.shape[1:]}; "
              f"clean={(regimes[mask]=='clean').sum()}, "
              f"contended={(regimes[mask]=='contended').sum()}")

for t in ["sf", "mod", "beacon", "pkt"]:
    build_sequences(t)
\end{lstlisting}

\subsection{Step 2 --- CNN-1D model}

\begin{lstlisting}[language=Python, caption={\texttt{train\_cnn.py} --- small 1-D CNN with global average pooling.}]
import torch, torch.nn as nn, numpy as np
from torch.utils.data import TensorDataset, DataLoader

class CNN1D(nn.Module):
    def __init__(self, n_features, n_classes):
        super().__init__()
        self.net = nn.Sequential(
            # Input is (B, F, T); Conv1d convolves along T
            nn.Conv1d(n_features, 16, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.Conv1d(16, 32, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.AdaptiveAvgPool1d(1),   # global average pool over T
            nn.Flatten(),
            nn.Linear(32, 32), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(32, n_classes),
        )

    def forward(self, x):
        # x comes in as (B, T, F); swap to (B, F, T)
        x = x.transpose(1, 2)
        return self.net(x)

def train_cnn(task, epochs=200, patience=15, lr=1e-3, batch_size=32):
    Xs_tr = np.load(f"artifacts/{task}/Xs_train.npy")
    ys_tr = np.load(f"artifacts/{task}/ys_train.npy")
    Xs_va = np.load(f"artifacts/{task}/Xs_val.npy")
    ys_va = np.load(f"artifacts/{task}/ys_val.npy")
    Xs_te = np.load(f"artifacts/{task}/Xs_test.npy")
    ys_te = np.load(f"artifacts/{task}/ys_test.npy")

    n_classes = int(ys_tr.max()) + 1
    device = "cuda" if torch.cuda.is_available() else "cpu"
    model = CNN1D(Xs_tr.shape[2], n_classes).to(device)
    opt = torch.optim.Adam(model.parameters(), lr=lr)
    loss_fn = nn.CrossEntropyLoss()
    ds = TensorDataset(torch.from_numpy(Xs_tr), torch.from_numpy(ys_tr))
    ld = DataLoader(ds, batch_size=batch_size, shuffle=True)
    X_va_t = torch.from_numpy(Xs_va).to(device)
    y_va_t = torch.from_numpy(ys_va).to(device)

    best_val, best_state, bad = float("inf"), None, 0
    for ep in range(epochs):
        model.train()
        for xb, yb in ld:
            xb, yb = xb.to(device), yb.to(device)
            opt.zero_grad()
            loss = loss_fn(model(xb), yb)
            loss.backward(); opt.step()
        model.eval()
        with torch.no_grad():
            v_logits = model(X_va_t)
            v_loss = loss_fn(v_logits, y_va_t).item()
        if v_loss < best_val - 1e-4:
            best_val = v_loss; best_state = {k: v.clone()
                for k, v in model.state_dict().items()}; bad = 0
        else:
            bad += 1
            if bad >= patience: break

    model.load_state_dict(best_state)
    model.eval()
    with torch.no_grad():
        X_te_t = torch.from_numpy(Xs_te).to(device)
        y_te_t = torch.from_numpy(ys_te).to(device)
        acc = (model(X_te_t).argmax(1) == y_te_t).float().mean().item()
    torch.save({"state_dict": model.state_dict(),
                "n_features": Xs_tr.shape[2], "n_classes": n_classes,
                "test_acc": acc}, f"artifacts/{task}/cnn.pt")
    print(f"[{task}] CNN test acc = {acc:.3f}")
    return model, acc

for t in ["sf", "mod", "beacon", "pkt"]:
    train_cnn(t)
\end{lstlisting}

\subsection{Step 3 --- The clean$\to$contended operating-condition shift experiment}
Define:
\begin{itemize}[leftmargin=2em]
	\item $\text{Acc}(\text{clean}\!\to\!\text{clean})$: model trained on clean-regime rows only (C1--C4), evaluated on the held-out clean-regime test split.
	\item $\text{Acc}(\text{clean}\!\to\!\text{contended})$: same clean-trained model, evaluated on the held-out contended-regime test split (C5).
	\item The operating-condition gap: $\Delta_{\text{op-shift}} = \text{Acc}(\text{clean}\!\to\!\text{clean}) - \text{Acc}(\text{clean}\!\to\!\text{contended})$.
\end{itemize}

A larger $\Delta_{\text{op-shift}}$ means the contended regime introduces feature-distribution changes the clean-trained model cannot handle (most likely from half-duplex shadowing, ACK exchanges, and Threat PAUSE bursts). Report $\Delta_{\text{op-shift}}$ for each of the four tasks; the packet-type task is expected to show the largest gap because the contended regime is the only one containing ACK and PAUSE rows.

\begin{lstlisting}[language=Python, caption={Operating-condition shift evaluation --- full reference implementation.}]
import numpy as np
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, TensorDataset

DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")

def _load_regime_split(task, regime_filter, split_filter):
    """Load CNN-ready windows filtered by regime and split.

    regime_filter: 'clean' or 'contended' (matches the regs_*.npy column)
    split_filter:  'train', 'val', or 'test'
    """
    Xs = np.load(f"artifacts/{task}/Xs_{split_filter}.npy")
    ys = np.load(f"artifacts/{task}/ys_{split_filter}.npy")
    rs = np.load(f"artifacts/{task}/regs_{split_filter}.npy")
    mask = (rs == regime_filter)
    return Xs[mask], ys[mask]

def _train(model, X, y, epochs=20, batch=64, lr=1e-3):
    model = model.to(DEVICE)
    ds = TensorDataset(torch.from_numpy(X).float(),
                       torch.from_numpy(y).long())
    dl = DataLoader(ds, batch_size=batch, shuffle=True)
    opt = torch.optim.Adam(model.parameters(), lr=lr)
    loss_fn = nn.CrossEntropyLoss()
    model.train()
    for _ in range(epochs):
        for xb, yb in dl:
            xb, yb = xb.to(DEVICE), yb.to(DEVICE)
            opt.zero_grad()
            loss_fn(model(xb), yb).backward()
            opt.step()
    return model

@torch.no_grad()
def _accuracy(model, X, y):
    model.eval()
    if len(X) == 0:
        return float("nan")
    xb = torch.from_numpy(X).float().to(DEVICE)
    yb = torch.from_numpy(y).long().to(DEVICE)
    pred = model(xb).argmax(dim=1)
    return (pred == yb).float().mean().item()

def op_shift(task):
    """Train on clean-regime rows only, eval on both regimes' test splits.

    Note: the scaler used to build Xs_*.npy was fit on the full training
    split in LH4-B Step 4, which slightly leaks contended-regime statistics
    into the clean-trained model's input distribution. The leak is small
    (StandardScaler statistics are scalar mean/var, not per-row), and the
    fix --- refitting the scaler on clean-only training data --- requires
    rebuilding the artifacts. Document either choice in your report.
    """
    # 1. Train on the clean-regime training rows
    X_tr, y_tr = _load_regime_split(task, "clean", "train")
    n_classes = int(y_tr.max()) + 1
    model = CNN1D(X_tr.shape[2], n_classes)
    model = _train(model, X_tr, y_tr)

    # 2. Evaluate on each regime's held-out test split
    X_clean,    y_clean    = _load_regime_split(task, "clean",     "test")
    X_contended, y_contended = _load_regime_split(task, "contended", "test")
    acc_clean    = _accuracy(model, X_clean,    y_clean)
    acc_contend  = _accuracy(model, X_contended, y_contended)
    delta = acc_clean - acc_contend
    return model, acc_clean, acc_contend, delta

results = []
for task in ["sf", "mod", "beacon", "pkt"]:
    _, acc_clean, acc_contend, delta = op_shift(task)
    results.append((task, acc_clean, acc_contend, delta))
    print(f"[{task}] clean->clean = {acc_clean:.3f}, "
          f"clean->contended = {acc_contend:.3f}, "
          f"Delta = {delta:.3f}")
\end{lstlisting}

\begin{tcolorbox}[warning={Why On-Hardware Operating-Condition Shift, Not Sim-to-Real?}]
	The framework deliberately measures distribution shift within hardware data because that is a more honest test of generalisation than synthetic-vs-real. The C5 regime introduces real RF-domain effects --- half-duplex shadowing during ACK exchanges, multi-station collisions under ALOHA, and Threat PAUSE bursts that produce non-stationary inter-arrival timing --- that no simulator could reproduce faithfully. The gap measured here is therefore directly attributable to physical operating-condition changes, not to simulator-vs-hardware modelling artefacts. This positioning is also consistent with the final report's framing (Section~IV.J): the operating-condition shift is the framework's capstone observation, and it is performed entirely on hardware data.
\end{tcolorbox}

\subsection{Step 5 --- Mitigation: fine-tuning the final layer on contended data}
Take the model trained on clean-regime data, freeze all layers except the final \texttt{Linear}, and fine-tune on a small fraction (10--20\%) of the contended-regime training set. Measure the new accuracy on the contended-regime test set. Expected outcome: most of the operating-condition gap closes, demonstrating that the convolutional backbone learned useful representations but the classifier head was mis-calibrated to the clean-regime feature distribution. The fine-tuning recovery is the framework-required outcome on the SF task: at least half of $\Delta_{\text{op-shift}}$ should be recovered.

\begin{lstlisting}[language=Python]
def finetune_last_layer(task, clean_model, fraction=0.10,
                        epochs=5, batch=32, lr=1e-4):
    # Freeze backbone, train only the final Linear layer
    for p in clean_model.parameters():       p.requires_grad = False
    for p in clean_model.net[-1].parameters(): p.requires_grad = True

    X_ft, y_ft = _load_regime_split(task, "contended", "train")
    if len(X_ft) == 0:
        print(f"[{task}] no contended training rows; skipping fine-tune")
        return float("nan")
    rng = np.random.default_rng(seed=42)
    keep = rng.choice(len(X_ft), size=int(fraction * len(X_ft)),
                      replace=False)
    X_ft, y_ft = X_ft[keep], y_ft[keep]

    clean_model = clean_model.to(DEVICE)
    ds = TensorDataset(torch.from_numpy(X_ft).float(),
                       torch.from_numpy(y_ft).long())
    dl = DataLoader(ds, batch_size=batch, shuffle=True)
    opt = torch.optim.Adam([p for p in clean_model.parameters()
                            if p.requires_grad], lr=lr)
    loss_fn = nn.CrossEntropyLoss()
    clean_model.train()
    for _ in range(epochs):
        for xb, yb in dl:
            xb, yb = xb.to(DEVICE), yb.to(DEVICE)
            opt.zero_grad()
            loss_fn(clean_model(xb), yb).backward()
            opt.step()

    # Evaluate on the contended held-out test split
    X_ev, y_ev = _load_regime_split(task, "contended", "test")
    return _accuracy(clean_model, X_ev, y_ev)

# Run for each task, reusing the clean-trained model from op_shift above
for task in ["sf", "mod", "beacon", "pkt"]:
    clean_model, acc_clean, acc_contend, delta = op_shift(task)
    acc_finetune = finetune_last_layer(task, clean_model)
    print(f"[{task}] clean+contended-finetune test acc = "
          f"{acc_finetune:.3f}  (was {acc_contend:.3f}, "
          f"recovered {acc_finetune - acc_contend:+.3f})")
\end{lstlisting}
