\labheader{1-H}{\FPGA{} \UART{} Ingestion and End-to-End Data Flow}

\section{Objective}
Implement a minimal \UART{} ingestion path on the Nexys A7-100T that receives the 43-byte forwarding frames from the STM32 interceptor (LH1-G), detects the start delimiter, extracts the Beacon~\mbox{ID}, and displays it on the seven-segment display. This closes the Module~1 end-to-end data path: \textbf{cooperative target $\to$ \RF{} link $\to$ interceptor (Threat) $\to$ \FPGA{}}.

\section{Prerequisites}
\begin{itemize}[leftmargin=2em]
	\item Lab Handouts~1-E (Vivado toolchain), 1-F (cooperative target firmware), and 1-G (interceptor / Threat firmware) complete.
	\item The interceptor node from LH1-G must be producing valid frames on \texttt{USART3 TX} (\texttt{PC4}). If you skipped LH1-G Step~7 because no USB-UART adapter was available, this handout is the validation step --- the FPGA seven-segment display will count frames in lieu of the host-side parser.
	\item LH1-E loopback projects \emph{do not} need to be open; this handout creates a separate Vivado project (\texttt{ingest\_top}) that lives alongside them.
\end{itemize}

\section{Wiring}
Connect the Threat's \texttt{USART3 TX} pin to one of the Nexys A7 \mbox{PMOD} input pins. Common ground is essential.

\begin{table}[h!]
	\centering
	\begin{tabular}{l l l l}
		\toprule
		\textbf{From (Threat Nucleo)} & \textbf{Function} & \textbf{To (Nexys A7)} & \textbf{FPGA Pin} \\
		\midrule
		Morpho \texttt{PC4} (USART3\_TX) & 3.3\,V serial out, 115200\,8N1 & PMOD JA pin 1 (\texttt{JA[0]}) & \texttt{C17} \\
		Morpho \texttt{GND}              & Common ground                  & PMOD JA pin 5 (or 11)         & --- \\
		\bottomrule
	\end{tabular}
\end{table}

\labfigure{LH1H_pmod_wiring.png}{0.8\textwidth}{STM32 \texttt{USART3 TX} (\texttt{PC4}) wired to Nexys A7 PMOD JA pin 1 (FPGA pin \texttt{C17}, called \texttt{JA[0]} in the master XDC) with common ground on PMOD JA pin 5. Annotated photo clarifies which PMOD connector is in use.}{fig:lh1h-pmod}

\begin{tcolorbox}[warning={Voltage Levels and Connector Choice}]
	The STM32 drives 3.3\,V logic and the Nexys A7 PMOD pins are 3.3\,V native. No level shifter is required. \emph{Use PMOD JA, not JC} --- some Nexys A7 board revisions wire JC (the high-speed PMOD next to the FMC connector) to a different I/O bank, and the LVDS-style pin pairs there have additional series resistors that can corrupt single-ended UART. JA, JB, and JD are all safe.
\end{tcolorbox}

\begin{lstlisting}[language=VHDL, caption={\texttt{uart\_rx.vhd} --- 115200 8N1 receiver, 16$\times$ oversampled. Generic \texttt{CLK\_FREQ\_HZ} and \texttt{BAUD} let you reuse the block at other rates.}]
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.NUMERIC_STD.ALL;

entity uart_rx is
    generic (
        CLK_FREQ_HZ : integer := 100_000_000;
        BAUD        : integer := 115_200
    );
    port (
        clk        : in  std_logic;
        rst_n      : in  std_logic;
        rx         : in  std_logic;
        rx_data    : out std_logic_vector(7 downto 0);
        rx_strobe  : out std_logic
    );
end uart_rx;

architecture rtl of uart_rx is
    constant DIV16 : integer := CLK_FREQ_HZ / (BAUD * 16);  -- ~54 at 100 MHz
    signal os_cnt : unsigned(15 downto 0) := (others => '0');
    signal os_tick : std_logic := '0';

    type state_t is (S_IDLE, S_START, S_DATA, S_STOP);
    signal state    : state_t := S_IDLE;
    signal bit_idx  : unsigned(2 downto 0) := (others => '0');
    signal sub_cnt  : unsigned(3 downto 0) := (others => '0');
    signal shifter  : std_logic_vector(7 downto 0) := (others => '0');
    signal rx_sync1, rx_sync2 : std_logic := '1';
begin
    -- Two-flop synchroniser for the async rx pin
    process (clk) begin
        if rising_edge(clk) then
            rx_sync1 <= rx;
            rx_sync2 <= rx_sync1;
        end if;
    end process;

    -- 16x oversample tick generator
    process (clk) begin
        if rising_edge(clk) then
            if rst_n = '0' then
                os_cnt  <= (others => '0');
                os_tick <= '0';
            elsif os_cnt = to_unsigned(DIV16 - 1, os_cnt'length) then
                os_cnt  <= (others => '0');
                os_tick <= '1';
            else
                os_cnt  <= os_cnt + 1;
                os_tick <= '0';
            end if;
        end if;
    end process;

    -- Receive FSM, advanced on os_tick
    process (clk) begin
        if rising_edge(clk) then
            rx_strobe <= '0';
            if rst_n = '0' then
                state   <= S_IDLE;
                sub_cnt <= (others => '0');
                bit_idx <= (others => '0');
            elsif os_tick = '1' then
                case state is
                    when S_IDLE =>
                        if rx_sync2 = '0' then           -- start bit
                            state   <= S_START;
                            sub_cnt <= to_unsigned(0, 4);
                        end if;
                    when S_START =>
                        if sub_cnt = to_unsigned(7, 4) then  -- mid-start
                            if rx_sync2 = '0' then       -- still low
                                state   <= S_DATA;
                                sub_cnt <= (others => '0');
                                bit_idx <= (others => '0');
                            else
                                state <= S_IDLE;          -- glitch
                            end if;
                        else
                            sub_cnt <= sub_cnt + 1;
                        end if;
                    when S_DATA =>
                        if sub_cnt = to_unsigned(15, 4) then
                            shifter <= rx_sync2 & shifter(7 downto 1);
                            sub_cnt <= (others => '0');
                            if bit_idx = to_unsigned(7, 3) then
                                state <= S_STOP;
                            else
                                bit_idx <= bit_idx + 1;
                            end if;
                        else
                            sub_cnt <= sub_cnt + 1;
                        end if;
                    when S_STOP =>
                        if sub_cnt = to_unsigned(15, 4) then
                            rx_data   <= shifter;
                            rx_strobe <= '1';
                            state     <= S_IDLE;
                        else
                            sub_cnt <= sub_cnt + 1;
                        end if;
                end case;
            end if;
        end if;
    end process;
end rtl;
\end{lstlisting}

\subsubsection*{Step 2.2 --- Create \texttt{byte\_fifo.vhd}}

Repeat the Add Sources $\to$ Create File flow with \textbf{File name: \texttt{byte\_fifo}}. Replace contents with:

\begin{lstlisting}[language=VHDL, caption={\texttt{byte\_fifo.vhd} --- 16-entry $\times$ 8-bit synchronous FIFO. Hand-rolled (no Xilinx primitive) so the file is portable to other parts.}]
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.NUMERIC_STD.ALL;

entity byte_fifo is
    generic (DEPTH_LOG2 : integer := 4);   -- 2**4 = 16 entries
    port (
        clk      : in  std_logic;
        rst_n    : in  std_logic;
        wr_en    : in  std_logic;
        wr_data  : in  std_logic_vector(7 downto 0);
        rd_en    : in  std_logic;
        rd_data  : out std_logic_vector(7 downto 0);
        empty    : out std_logic;
        full     : out std_logic
    );
end byte_fifo;

architecture rtl of byte_fifo is
    constant DEPTH : integer := 2**DEPTH_LOG2;
    type   mem_t is array (0 to DEPTH - 1) of std_logic_vector(7 downto 0);
    signal mem    : mem_t := (others => (others => '0'));
    signal w_ptr  : unsigned(DEPTH_LOG2 downto 0) := (others => '0');
    signal r_ptr  : unsigned(DEPTH_LOG2 downto 0) := (others => '0');
    signal w_idx  : unsigned(DEPTH_LOG2 - 1 downto 0);
    signal r_idx  : unsigned(DEPTH_LOG2 - 1 downto 0);
begin
    w_idx <= w_ptr(DEPTH_LOG2 - 1 downto 0);
    r_idx <= r_ptr(DEPTH_LOG2 - 1 downto 0);
    empty <= '1' when w_ptr = r_ptr else '0';
    full  <= '1' when (w_ptr(DEPTH_LOG2) /= r_ptr(DEPTH_LOG2)) and (w_idx = r_idx) else '0';

    process (clk) begin
        if rising_edge(clk) then
            if rst_n = '0' then
                w_ptr <= (others => '0');
                r_ptr <= (others => '0');
            else
                if wr_en = '1' and (w_ptr(DEPTH_LOG2) = r_ptr(DEPTH_LOG2) or w_idx /= r_idx) then
                    mem(to_integer(w_idx)) <= wr_data;
                    w_ptr <= w_ptr + 1;
                end if;
                if rd_en = '1' and w_ptr /= r_ptr then
                    r_ptr <= r_ptr + 1;
                end if;
            end if;
            rd_data <= mem(to_integer(r_idx));
        end if;
    end process;
end rtl;
\end{lstlisting}

\subsubsection*{Step 2.3 --- Create \texttt{frame\_detector.vhd}}

Repeat the Add Sources flow with \textbf{File name: \texttt{frame\_detector}}. Replace contents with:

\begin{lstlisting}[language=VHDL, caption={\texttt{frame\_detector.vhd} --- IDLE/LEN/PAYLOAD/DONE FSM. Captures the radio-packet's Beacon~ID (frame offset 12) into \texttt{beacon\_id} and pulses \texttt{new\_frame} once per valid frame.}]
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.NUMERIC_STD.ALL;

entity frame_detector is
    port (
        clk         : in  std_logic;
        rst_n       : in  std_logic;
        fifo_empty  : in  std_logic;
        fifo_data   : in  std_logic_vector(7 downto 0);
        fifo_rd_en  : out std_logic;
        beacon_id   : out std_logic_vector(7 downto 0);
        new_frame   : out std_logic
    );
end frame_detector;

architecture rtl of frame_detector is
    type state_t is (S_IDLE, S_LEN, S_PAYLOAD, S_DONE);
    signal state    : state_t := S_IDLE;
    signal byte_cnt : unsigned(5 downto 0) := (others => '0');  -- 0..40
    signal id_reg   : std_logic_vector(7 downto 0) := (others => '0');
begin
    process (clk) begin
        if rising_edge(clk) then
            fifo_rd_en <= '0';
            new_frame  <= '0';

            if rst_n = '0' then
                state    <= S_IDLE;
                byte_cnt <= (others => '0');
            elsif fifo_empty = '0' then
                fifo_rd_en <= '1';      -- pop one byte this cycle
                case state is
                    when S_IDLE =>
                        if fifo_data = x"7E" then
                            state <= S_LEN;
                        end if;
                    when S_LEN =>
                        if fifo_data = x"29" then        -- length = 41
                            state    <= S_PAYLOAD;
                            byte_cnt <= (others => '0');
                        else
                            state <= S_IDLE;             -- bad length
                        end if;
                    when S_PAYLOAD =>
                        -- The 41 payload bytes are frame[1..41].  We have already
                        -- consumed frame[1] (length).  We now consume the 40
                        -- remaining bytes 2..41.  The Beacon~ID lives at frame
                        -- offset 12 = radio-packet byte 2; the first PAYLOAD
                        -- byte we see here is frame offset 2 (timestamp byte 0),
                        -- so we capture when byte_cnt = 10 (12 - 2).
                        if byte_cnt = to_unsigned(10, 6) then
                            id_reg <= fifo_data;
                        end if;
                        if byte_cnt = to_unsigned(39, 6) then  -- last byte (frame[41])
                            state <= S_DONE;
                        else
                            byte_cnt <= byte_cnt + 1;
                        end if;
                    when S_DONE =>
                        beacon_id <= id_reg;
                        new_frame <= '1';
                        state     <= S_IDLE;
                        fifo_rd_en <= '0';   -- skip the trailing CRC-8 byte? no:
                                             -- the CRC-8 byte (frame[42]) will be
                                             -- silently consumed next cycle by
                                             -- IDLE-state non-0x7E discard.
                end case;
            end if;
        end if;
    end process;
end rtl;
\end{lstlisting}

\begin{tcolorbox}[infobox={Why \texttt{S\_IDLE} silently discards}]
The frame CRC-8 byte at frame offset 42 is not validated in Module~1 --- the IDLE state's ``pop and check for 0x7E'' behavior naturally discards any byte that is not a delimiter, so a valid CRC byte that happens not to be 0x7E falls through harmlessly. If the CRC byte happens to equal 0x7E (which occurs $\sim$0.4\,\% of frames), the FSM mis-aligns by one frame; on the next bad-length byte at S\_LEN it resyncs. This cost is acceptable for the LH1-H end-to-end demo. Module~2 LH2-D adds a proper CRC check and an explicit consume-and-discard byte at the end of each frame.
\end{tcolorbox}

\subsubsection*{Step 2.4 --- Create \texttt{seg7\_driver.vhd}}

Repeat the Add Sources flow with \textbf{File name: \texttt{seg7\_driver}}. Replace contents with:

\begin{lstlisting}[language=VHDL, caption={\texttt{seg7\_driver.vhd} --- multiplexes the four right-most digits at $\sim$1\,kHz per digit. Right two digits show the live Beacon~ID; left two digits show a 16-bit frame counter modulo 256.}]
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.NUMERIC_STD.ALL;

entity seg7_driver is
    generic (CLK_FREQ_HZ : integer := 100_000_000);
    port (
        clk        : in  std_logic;
        rst_n      : in  std_logic;
        beacon_id  : in  std_logic_vector(7 downto 0);
        new_frame  : in  std_logic;
        seg        : out std_logic_vector(6 downto 0);   -- a..g, active-low
        dp         : out std_logic;                       -- decimal point, active-low
        an         : out std_logic_vector(7 downto 0)     -- digit anodes, active-low
    );
end seg7_driver;

architecture rtl of seg7_driver is
    -- 4 kHz refresh => 1 kHz per digit
    constant REFRESH_DIV : integer := CLK_FREQ_HZ / 4_000;
    signal ref_cnt : unsigned(15 downto 0) := (others => '0');
    signal digit_sel : unsigned(1 downto 0) := (others => '0');
    signal frame_ctr : unsigned(15 downto 0) := (others => '0');
    signal nibble    : std_logic_vector(3 downto 0);

    function hex7(n : std_logic_vector(3 downto 0)) return std_logic_vector is
    begin
        case n is
            when x"0" => return "1000000";  when x"1" => return "1111001";
            when x"2" => return "0100100";  when x"3" => return "0110000";
            when x"4" => return "0011001";  when x"5" => return "0010010";
            when x"6" => return "0000010";  when x"7" => return "1111000";
            when x"8" => return "0000000";  when x"9" => return "0010000";
            when x"A" => return "0001000";  when x"B" => return "0000011";
            when x"C" => return "1000110";  when x"D" => return "0100001";
            when x"E" => return "0000110";  when x"F" => return "0001110";
            when others => return "1111111";
        end case;
    end function;
begin
    dp <= '1';                           -- decimal point off

    -- Frame counter: increments on every new_frame pulse
    process (clk) begin
        if rising_edge(clk) then
            if rst_n = '0' then
                frame_ctr <= (others => '0');
            elsif new_frame = '1' then
                frame_ctr <= frame_ctr + 1;
            end if;
        end if;
    end process;

    -- Digit-select counter
    process (clk) begin
        if rising_edge(clk) then
            if rst_n = '0' then
                ref_cnt <= (others => '0');
                digit_sel <= (others => '0');
            elsif ref_cnt = to_unsigned(REFRESH_DIV - 1, ref_cnt'length) then
                ref_cnt <= (others => '0');
                digit_sel <= digit_sel + 1;
            else
                ref_cnt <= ref_cnt + 1;
            end if;
        end if;
    end process;

    -- Choose nibble per digit and drive anodes
    process (digit_sel, beacon_id, frame_ctr) begin
        an <= (others => '1');           -- all off (active-low)
        case digit_sel is
            when "00" =>                 -- AN0: low nibble of beacon_id
                nibble <= beacon_id(3 downto 0);
                an(0) <= '0';
            when "01" =>                 -- AN1: high nibble of beacon_id
                nibble <= beacon_id(7 downto 4);
                an(1) <= '0';
            when "10" =>                 -- AN2: low nibble of frame_ctr
                nibble <= std_logic_vector(frame_ctr(3 downto 0));
                an(2) <= '0';
            when others =>               -- AN3: next nibble of frame_ctr
                nibble <= std_logic_vector(frame_ctr(7 downto 4));
                an(3) <= '0';
        end case;
    end process;

    seg <= hex7(nibble);
end rtl;
\end{lstlisting}

\subsubsection*{Step 2.5 --- Create \texttt{ingest\_top.vhd} and set as top}

Repeat the Add Sources flow one last time with \textbf{File name: \texttt{ingest\_top}}. Replace contents with:

\begin{lstlisting}[language=VHDL, caption={\texttt{ingest\_top.vhd} --- top-level wiring. After saving, right-click \texttt{ingest\_top} in the Sources panel and choose \textbf{Set as Top}; Vivado will mark it bold.}]
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity ingest_top is
    port (
        clk100  : in  std_logic;
        cpu_resetn : in  std_logic;          -- active-low pushbutton
        uart_rx_pin : in  std_logic;         -- from Threat USART3 TX
        seg     : out std_logic_vector(6 downto 0);
        dp      : out std_logic;
        an      : out std_logic_vector(7 downto 0);
        led     : out std_logic_vector(15 downto 0)
    );
end ingest_top;

architecture rtl of ingest_top is
    signal rst_n      : std_logic;
    signal rx_data    : std_logic_vector(7 downto 0);
    signal rx_strobe  : std_logic;
    signal fifo_empty : std_logic;
    signal fifo_full  : std_logic;
    signal fifo_rd    : std_logic;
    signal fifo_q     : std_logic_vector(7 downto 0);
    signal beacon_id  : std_logic_vector(7 downto 0) := (others => '0');
    signal new_frame  : std_logic;
begin
    rst_n <= cpu_resetn;

    -- Heartbeat / status LEDs
    led(0)  <= rx_strobe;
    led(1)  <= new_frame;
    led(2)  <= fifo_empty;
    led(3)  <= fifo_full;
    led(15 downto 4) <= (others => '0');

    u_uart : entity work.uart_rx
        generic map (CLK_FREQ_HZ => 100_000_000, BAUD => 115_200)
        port map (
            clk => clk100, rst_n => rst_n, rx => uart_rx_pin,
            rx_data => rx_data, rx_strobe => rx_strobe
        );

    u_fifo : entity work.byte_fifo
        port map (
            clk => clk100, rst_n => rst_n,
            wr_en => rx_strobe, wr_data => rx_data,
            rd_en => fifo_rd, rd_data => fifo_q,
            empty => fifo_empty, full => fifo_full
        );

    u_det : entity work.frame_detector
        port map (
            clk => clk100, rst_n => rst_n,
            fifo_empty => fifo_empty, fifo_data => fifo_q,
            fifo_rd_en => fifo_rd,
            beacon_id => beacon_id, new_frame => new_frame
        );

    u_seg : entity work.seg7_driver
        generic map (CLK_FREQ_HZ => 100_000_000)
        port map (
            clk => clk100, rst_n => rst_n,
            beacon_id => beacon_id, new_frame => new_frame,
            seg => seg, dp => dp, an => an
        );
end rtl;
\end{lstlisting}

\subsection{Step 5 --- End-to-end run}

\begin{enumerate}[leftmargin=2em]
	\item Confirm the wiring per the table on page 2: Nucleo Morpho \texttt{PC4} $\to$ PMOD JA pin 1, GND $\to$ PMOD JA pin 5.
	\item Power on Target~A and Target~B (LH1-F). Confirm their USART2 logs show the ping-pong CSV lines.
	\item Power on the Threat (LH1-G). Confirm its USART2 log shows \texttt{RX sender=01} and \texttt{RX sender=02} lines alternating once per second.
	\item Observe the Nexys A7 seven-segment display. Right-most two digits should alternate \texttt{01} (Target~A's DATA) and \texttt{02} (Target~B's ACK) in step with each frame; left-most two digits should show a slowly-incrementing 16-bit frame counter (low byte). \texttt{LED[0]} (UART byte strobe) should flicker rapidly; \texttt{LED[1]} (new\_frame pulse) should blink at the per-frame cadence; \texttt{LED[2]} (FIFO empty) should be on most of the time; \texttt{LED[3]} (FIFO full) should never light.
	\item \textbf{Perturbation 1:} power off Target~A. The display should lock to \texttt{02} for one frame and then go silent (Target~B has nothing to ACK). Power Target~A back on; alternating display should resume within 2\,s.
	\item \textbf{Perturbation 2:} hold the Threat user button (B1) for 3\,s. The display should briefly fall silent (the targeted Target stops transmitting DATA, so the other has nothing to ACK), then resume alternating within $\sim$500\,ms of release.
\end{enumerate}

\section{Verification Checklist}
\begin{itemize}[leftmargin=2em]
	\item[$\square$] Vivado project \texttt{ingest\_top} created with all five VHDL sources and one XDC; \texttt{ingest\_top} marked as Top.
	\item[$\square$] Synthesis, Implementation, and Bitstream generation complete with zero errors.
	\item[$\square$] WNS $\geq 0$\,ns and WHS $\geq 0$\,ns in the implementation timing report.
	\item[$\square$] DONE LED illuminates after programming.
	\item[$\square$] \texttt{LED[0]} flickers (UART bytes arriving), \texttt{LED[1]} pulses at $\sim$1\,Hz (new\_frame), \texttt{LED[3]} (FIFO full) never lights.
	\item[$\square$] Seven-segment display alternates \texttt{01} and \texttt{02} matching the live Beacon~IDs.
	\item[$\square$] Powering Target~A off then on causes the display to silence then resume within 2\,s.
	\item[$\square$] Holding the Threat's user button suspends the alternating display; releasing it restores normal operation within $\sim$500\,ms.
\end{itemize}

\begin{tcolorbox}[warning={Troubleshooting}]
	\begin{itemize}[leftmargin=1.2em]
		\item \textbf{No display change, \texttt{LED[0]} dark:} no UART bytes are reaching the FPGA. Re-check the PMOD wire (JA pin 1, not pin 2) and the GND wire. Verify with a scope on the FPGA side of the jumper that you see 3.3\,V LVCMOS-level transitions when the Threat is transmitting.
		\item \textbf{\texttt{LED[0]} flickers but \texttt{LED[1]} stays dark:} bytes are arriving but no frame ever assembles. Likely the FIFO write or read enable is wrong, or the FSM never sees \texttt{0x7E}. Add an ILA probe on \texttt{rx\_data} and look for \texttt{0x7E} bytes; if absent, your Threat is not actually emitting frames (re-validate LH1-G Step~7 with the Python parser).
		\item \textbf{Display garbled / wrong digits:} segment-anode mapping inverted. The Nexys A7 segments and anodes are \emph{both active-low}; double-check your XDC and the \texttt{hex7} lookup table.
		\item \textbf{Display flickers unreadably:} multiplex rate too slow. The default REFRESH\_DIV gives 4\,kHz total / 1\,kHz per digit; if you reduced it, raise it back.
		\item \textbf{Frame detector stuck (\texttt{LED[1]} never pulses) but \texttt{0x7E} bytes are visible on ILA:} length byte mismatch. The Threat must be sending \texttt{0x29} as byte~1 of every frame. If you see anything else, your Threat firmware has the wrong length constant.
	\end{itemize}
\end{tcolorbox}
