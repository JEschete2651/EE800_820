\labheader{1-C}{\SPI{} Bring-Up: STM32 $\leftrightarrow$ SX1276 Register Access}

\section{Objective}
Establish reliable \SPI{} communication between the STM32 Nucleo-L476RG and the SX1276-based RFM95W module. You will read the \texttt{RegVersion} register and receive the expected value \texttt{0x12}, then demonstrate write/read-back and radio mode transitions. This handout is the first point at which firmware actually touches the \RF{} silicon.

\section{Prerequisites}
\begin{itemize}[leftmargin=2em]
	\item Lab Handout~1-A complete (breakout physically assembled).
	\item Lab Handout~1-B complete (toolchain working, blinky verified).
\end{itemize}

\section{Materials}
\begin{itemize}[leftmargin=2em]
	\item Assembled RFM95W breakout (from LH1-A).
	\item Nucleo-L476RG + USB cable (from LH1-B).
	\item Half-size breadboard, 8--10 male--female jumper wires.
	\item Logic analyzer (optional but strongly recommended for troubleshooting).
\end{itemize}

\begin{tcolorbox}[warning={Wire Before You Power}]
	Double-check \texttt{VIN} vs.\ \texttt{GND}. The RFM95W will tolerate 3.3--5\,V on \texttt{VIN} (it has an on-board regulator), but swapping \texttt{VIN} and \texttt{GND} \textit{will} destroy the module. Verify with a \mbox{DMM} before plugging the Nucleo into USB.
\end{tcolorbox}

\section{Procedure}

\subsection{Step 1 --- Configure peripherals via STM32CubeMX}
\begin{enumerate}[leftmargin=2em]
	\item Create a new project in STM32CubeMX following the same steps as Lab Handout~1-B (STM icon $\to$ Launch STM32CubeMX $\to$ Board Selector $\to$ NUCLEO-L476RG $\to$ Start Project $\to$ Project Manager). Name the project \textbf{Nucleo\_RFM95W\_Base} and save it to a local folder outside OneDrive. Set the toolchain to \textbf{CMake}. This project will serve as the reusable base configuration for all subsequent RF labs.

	\item \textbf{Enable SPI1:}
	\begin{enumerate}[leftmargin=1.5em]
		\item In the pinout view, click pin \texttt{PA5} and select \textbf{SPI1\_SCK} from the popup. Repeat for \texttt{PA6} $\to$ \textbf{SPI1\_MISO} and \texttt{PA7} $\to$ \textbf{SPI1\_MOSI}.
		\item In the left-hand \textbf{Categories} tree, expand \textbf{Connectivity} and click \textbf{SPI1}.
		\item Set the \textbf{Mode} dropdown to \textbf{Full-Duplex Master}. The pins assigned above will be confirmed in the pinout view.
	\end{enumerate}

	\item \textbf{Configure SPI1 parameters:}
	\begin{enumerate}[leftmargin=1.5em]
		\item With SPI1 still selected, click the \textbf{Configuration} tab (or \textbf{Parameter Settings} in newer CubeMX versions).
		\item Set each parameter per Table~\ref{tab:spi-bringup} below. Pay particular attention to \mbox{CPOL}=0 and \mbox{CPHA}=1~Edge --- an incorrect clock phase is the most common cause of a failed \texttt{RegVersion} read.
		\item Set \textbf{NSS} to \textbf{Software} --- chip select will be driven manually in firmware.
	\end{enumerate}

\begin{table}[h!]
	\centering
	\caption{\SPI{}1 configuration for RFM95W register access.}
	\label{tab:spi-bringup}
	\begin{tabular}{l l}
		\toprule
		\textbf{Parameter} & \textbf{Value} \\
		\midrule
		Mode & Full-Duplex Master \\
		Data size & 8 bits \\
		First bit & \mbox{MSB} first \\
		Prescaler & \textbf{32} (yields 2.5\,MHz at 80\,MHz PCLK2; safe margin below the 10\,MHz max) \\
		\mbox{CPOL} & 0 (Low) \\
		\mbox{CPHA} & 1 Edge \\
		NSS & Software \\
		\bottomrule
	\end{tabular}
\end{table}

	\item \textbf{Configure GPIO pins for CS, RST, and DIO0:}
	\begin{enumerate}[leftmargin=1.5em]
		\item In the pinout view, click pin \texttt{PB6}. Select \textbf{GPIO\_Output} from the popup.
		\item Repeat for \texttt{PA9} and \texttt{PA8}.
		\item To assign user labels, click \textbf{System Core} $\to$ \textbf{GPIO} in the categories tree. Select each pin and set its \textbf{User Label}: \texttt{PB6} $\to$ \texttt{RFM\_CS}, \texttt{PA9} $\to$ \texttt{RFM\_RST}, \texttt{PA8} $\to$ \texttt{RFM\_DIO0}. These labels become the \texttt{\_GPIO\_Port} and \texttt{\_Pin} macro names in generated code.
		\item Leave \texttt{PA8} (\texttt{RFM\_DIO0}) as a plain GPIO output for now; it will be reconfigured as an \texttt{EXTI} interrupt input in a later handout.
	\end{enumerate}

	\item \textbf{Enable USART2} (used to send \texttt{printf} output to the ST-LINK virtual COM port):
	\begin{enumerate}[leftmargin=1.5em]
		\item In the categories tree, expand \textbf{Connectivity} and click \textbf{USART2}.
		\item Set \textbf{Mode} to \textbf{Asynchronous}.
		\item In \textbf{Parameter Settings}, confirm \textbf{Baud Rate} is \textbf{115200}, \textbf{Word Length} 8~bits, \textbf{Parity} None, \textbf{Stop Bits} 1.
	\end{enumerate}

	\item Click \textbf{Generate Code}. Return to VS Code and run \texttt{cube bundle init -{}-project} in the terminal if prompted.
\end{enumerate}

\subsection{Step 2 --- Redirect \texttt{printf} to USART2}
\begin{enumerate}[leftmargin=2em]
	\item In \texttt{main.c}, locate the \texttt{/* USER CODE BEGIN Includes */} section near the top of the file and add:
\begin{verbatim}
/* USER CODE BEGIN Includes */
#include <stdio.h>
/* USER CODE END Includes */
\end{verbatim}
	\item Locate the \texttt{/* USER CODE BEGIN PFP */} section and add prototypes for every function you will define in this lab:
\begin{verbatim}
/* USER CODE BEGIN PFP */
int __io_putchar(int ch);
uint8_t rfm_read_reg(uint8_t addr);
void rfm_write_reg(uint8_t addr, uint8_t val);
/* USER CODE END PFP */
\end{verbatim}
	\item Scroll to the \texttt{/* USER CODE BEGIN 4 */} section near the bottom of the file and add the definition:
\begin{verbatim}
/* USER CODE BEGIN 4 */
int __io_putchar(int ch) {
    HAL_UART_Transmit(&huart2, (uint8_t*)&ch, 1, HAL_MAX_DELAY);
    return ch;
}
/* USER CODE END 4 */
\end{verbatim}
	Keeping the prototype and definition inside the \texttt{USER CODE} comment blocks ensures they are preserved if you regenerate code from CubeMX.
\end{enumerate}

\subsection{Step 3 --- Write a minimal register-read function}
Implement the following helper:
\begin{verbatim}
uint8_t rfm_read_reg(uint8_t addr) {
    uint8_t tx[2] = { addr & 0x7F, 0x00 };  // bit 7 = 0 => read
    uint8_t rx[2] = { 0, 0 };
    HAL_GPIO_WritePin(RFM_CS_GPIO_Port, RFM_CS_Pin, GPIO_PIN_RESET);
    HAL_SPI_TransmitReceive(&hspi1, tx, rx, 2, HAL_MAX_DELAY);
    HAL_GPIO_WritePin(RFM_CS_GPIO_Port, RFM_CS_Pin, GPIO_PIN_SET);
    return rx[1];
}
\end{verbatim}
Add the write helper immediately after in the same \texttt{USER CODE BEGIN 4} section:
\begin{verbatim}
void rfm_write_reg(uint8_t addr, uint8_t val) {
    uint8_t tx[2] = { addr | 0x80, val };  // bit 7 = 1 => write
    HAL_GPIO_WritePin(RFM_CS_GPIO_Port, RFM_CS_Pin, GPIO_PIN_RESET);
    HAL_SPI_Transmit(&hspi1, tx, 2, HAL_MAX_DELAY);
    HAL_GPIO_WritePin(RFM_CS_GPIO_Port, RFM_CS_Pin, GPIO_PIN_SET);
}
\end{verbatim}

\subsection{Step 4 --- Reset the module and read \texttt{RegVersion}}
\begin{enumerate}[leftmargin=2em]
	\item In \texttt{main.c}, locate the \texttt{/* USER CODE BEGIN 2 */} section (after all HAL and peripheral initialization). Add the following reset sequence and register read:
\begin{verbatim}
/* USER CODE BEGIN 2 */

/* Reset the RFM95W */
HAL_GPIO_WritePin(RFM_RST_GPIO_Port, RFM_RST_Pin, GPIO_PIN_RESET);
HAL_Delay(1);  // hold reset low for at least 100 us
HAL_GPIO_WritePin(RFM_RST_GPIO_Port, RFM_RST_Pin, GPIO_PIN_SET);
HAL_Delay(10); // wait for module to come out of reset

/* Read and print the version register */
uint8_t version = rfm_read_reg(0x42);
printf("RegVersion = 0x%02X\r\n", version);

/* USER CODE END 2 */
\end{verbatim}
	\item Build the project: Command Palette $\to$ \textbf{CMake: Build}. Confirm zero errors.
	\item Flash and run: in the \textbf{Run and Debug} panel, click the green \textbf{Start Debugging} arrow. Once the debugger halts at \texttt{main}, click the green \textbf{Continue} button in the debug toolbar.
	\item Open the Serial Monitor panel:
	\begin{enumerate}[leftmargin=1.5em]
		\item Command Palette (\texttt{Ctrl+Shift+P}) $\to$ \textbf{Serial Monitor: Focus on Serial Monitor View}.
		\item In the Serial Monitor panel at the bottom of VS Code, click \textbf{Open Serial Monitor}.
		\item Select the \textbf{STLink Virtual COM Port} from the port dropdown.
		\item Set the baud rate to \textbf{115200}.
		\item Click \textbf{Start Monitoring}.
	\end{enumerate}
	\item You should see the following output:
\begin{verbatim}
RegVersion = 0x12
\end{verbatim}
	Any other value indicates a wiring or configuration error --- see the Troubleshooting section.
\end{enumerate}

\begin{tcolorbox}[infobox={Why 0x12?}]
	The SX1276 silicon revision register returns \texttt{0x12} for the V1b mask set used in the RFM9x breakouts. This is the single most reliable sanity check for \SPI{} wiring, clock polarity, and reset sequencing: if you read \texttt{0x00} or \texttt{0xFF}, the module is not talking to you; if you read anything else, you have a clock phase or byte-order issue.
\end{tcolorbox}

\subsection{Step 5 --- Register write / read-back}

Add the following code in \texttt{main.c} inside the \texttt{/* USER CODE BEGIN 2 */} block, immediately after the \texttt{RegVersion} read from Step~4:

\begin{enumerate}[leftmargin=2em]
	\item First put the module into \texttt{SLEEP} mode --- the SX1276 requires this before frequency registers can be written:
\begin{verbatim}
rfm_write_reg(0x01, 0x80);  // LoRa mode, SLEEP
HAL_Delay(10);
\end{verbatim}
	\item Write the three frequency registers to set the carrier to 906.5\,MHz:
\begin{verbatim}
rfm_write_reg(0x06, 0xD9);  // RegFrfMsb
rfm_write_reg(0x07, 0x0C);  // RegFrfMid
rfm_write_reg(0x08, 0xCC);  // RegFrfLsb
\end{verbatim}
	\item Read the same registers back and print each value:
\begin{verbatim}
printf("FrfMsb = 0x%02X\r\n", rfm_read_reg(0x06));
printf("FrfMid = 0x%02X\r\n", rfm_read_reg(0x07));
printf("FrfLsb = 0x%02X\r\n", rfm_read_reg(0x08));
\end{verbatim}
	\item Build (CMake: Build), then flash using the green \textbf{Start Debugging} arrow in the \textbf{Run and Debug} panel. Check the serial monitor. The three printed values should match \texttt{0xD9}, \texttt{0x0C}, \texttt{0xCC}. A mismatch usually indicates a \mbox{CPOL}/\mbox{CPHA} error or a floating \texttt{MISO} line.
\end{enumerate}

\subsection{Step 6 --- Mode transitions}
The \texttt{RegOpMode} register (\texttt{0x01}) controls the radio operating state. The low 3~bits select the mode; bit~7 selects LoRa vs.\ FSK. Add the following to \texttt{/* USER CODE BEGIN 2 */} after the Step~5 code, then build and flash to observe each transition in the serial monitor.

\begin{enumerate}[leftmargin=2em]
	\item Read the current mode and print it:
\begin{verbatim}
printf("OpMode = 0x%02X\r\n", rfm_read_reg(0x01));
\end{verbatim}
	\item Transition to LoRa \texttt{SLEEP} (bits [2:0] = \texttt{000}):
\begin{verbatim}
rfm_write_reg(0x01, 0x80);
HAL_Delay(10);
printf("OpMode (SLEEP)    = 0x%02X\r\n", rfm_read_reg(0x01));
\end{verbatim}
	\item Transition to LoRa \texttt{STANDBY} (bits [2:0] = \texttt{001}):
\begin{verbatim}
rfm_write_reg(0x01, 0x81);
HAL_Delay(10);
printf("OpMode (STANDBY)  = 0x%02X\r\n", rfm_read_reg(0x01));
\end{verbatim}
	\item Transition to LoRa \texttt{TX} (bits [2:0] = \texttt{011}):
\begin{verbatim}
rfm_write_reg(0x01, 0x83);
HAL_Delay(10);
printf("OpMode (TX)       = 0x%02X\r\n", rfm_read_reg(0x01));
\end{verbatim}
	The module will return to \texttt{STANDBY} automatically after any transmission attempt; seeing \texttt{0x81} on a subsequent read is expected.
\end{enumerate}

\section{Verification Checklist}
\begin{itemize}[leftmargin=2em]
	\item[$\square$] \texttt{RegVersion} reads \texttt{0x12}.
	\item[$\square$] Write/read-back to \texttt{RegFrf*} registers returns identical values.
	\item[$\square$] \texttt{RegOpMode} mode bits transition correctly through \texttt{SLEEP}, \texttt{STANDBY}, \texttt{TX}.
	\item[$\square$] Serial output shows all register values in hex over the ST-LINK virtual COM port.
\end{itemize}

\begin{tcolorbox}[warning={Troubleshooting}]
	\begin{itemize}[leftmargin=1.2em]
		\item \textbf{\texttt{RegVersion} reads \texttt{0x00}}: \texttt{MISO} not connected, or \texttt{CS} never asserted.
		\item \textbf{\texttt{RegVersion} reads \texttt{0xFF}}: \texttt{MISO} floating (pull-up). Check the jumper to \texttt{PA6}.
		\item \textbf{Reads a seemingly-random but repeatable value}: \mbox{CPOL}/\mbox{CPHA} wrong, or bit 7 (read/write) flipped in your address byte.
		\item \textbf{First read is wrong, subsequent reads correct}: missing reset delay; increase the post-reset wait to 20\,ms.
	\end{itemize}
\end{tcolorbox}
